264   def self.solrize( input_hash, solr_doc=Solr::Document.new )    
265     facets = input_hash.has_key?(:facets) ? input_hash[:facets] : input_hash
 
266     facets.each_pair do |facet_name, value|
 
267       case value.class.to_s
 
268       when "String"
 
269         solr_doc << Solr::Field.new( :"#{facet_name}_facet" => "#{value}" )
 
270       when "Array"
 
271         value.each { |v| solr_doc << Solr::Field.new( :"#{facet_name}_facet" => "#{v}" ) } 
272       end
 
273     end
 
274     
275     if input_hash.has_key?(:symbols) 
276       input_hash[:symbols].each do |symbol_name, value|
 
277         case value.class.to_s
